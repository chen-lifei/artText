createBaseWatermark(wrapper, text1, text2, showText2 = false){
        if (document.getElementById('watermark') !== null) {
            document.getElementById('watermark').remove();
        }
        const canvasEle = document.createElement('canvas');
        const width = wrapper.scrollWidth;
        const height = wrapper.scrollHeight;
        canvasEle.width = width;
        canvasEle.height = height;
        const ctx = canvasEle.getContext('2d');
        ctx.rotate(-45 * Math.PI / 180);
        ctx.font = "34px '微软雅黑'";
        ctx.strokeStyle = 'rgba(170, 170, 170, 0.15)';
        ctx.textAlign = 'left';
        let markWidth1 = ctx.measureText(text1).width;
        let markWidth2 = ctx.measureText(text2).width;
        let content = text1 || visitorName;
        let markWidth = text1 ? markWidth1 : markWidth2;
        let count = 1;
        for (let j = -height / 2; j < height * 5; j += 150) {
            if (text1 && (!showText2 || count % 2 > 0)) {
                content = text1;
                markWidth = markWidth1;
            } else {
                content = text2;
                markWidth = markWidth2;
            }
            for (let i = -width / 2; i < width * 1.5; i += markWidth + 180) {
                ctx.strokeText(content, i, j);
            }
            count++;
        }
        // 转为图片
        const divEle = document.createElement('div');
        divEle.id = 'watermark';
        divEle.style.width = `${width}px`;
        divEle.style.height = `${height}px`;
        divEle.style.pointerEvents = 'none';
        divEle.style.zIndex = 2;
        divEle.style.position = 'absolute';
        divEle.style.top = 0;
        divEle.style.left = 0;
        divEle.style.background = 'url(' + canvasEle.toDataURL() + ') left top repeat';
        wrapper.appendChild(divEle);
    },
